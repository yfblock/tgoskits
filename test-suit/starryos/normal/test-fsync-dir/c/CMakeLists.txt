cmake_minimum_required(VERSION 3.20)
project(test-fsync-dir C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-fsync-dir src/main.c)
target_include_directories(test-fsync-dir PRIVATE src)
target_compile_options(test-fsync-dir PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-fsync-dir RUNTIME DESTINATION usr/bin)
