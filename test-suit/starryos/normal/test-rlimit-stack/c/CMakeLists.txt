cmake_minimum_required(VERSION 3.20)
project(test-rlimit-stack C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)
set(CMAKE_C_EXTENSIONS OFF)
add_executable(test-rlimit-stack src/main.c)
target_include_directories(test-rlimit-stack PRIVATE src)
target_compile_options(test-rlimit-stack PRIVATE -Wall -Wextra -Werror)
install(TARGETS test-rlimit-stack RUNTIME DESTINATION usr/bin)
